void *calloc(size_t nMember, size_t size)
{
    size_t buffSize = nMember * size;
    void *ptr = NULL;

    if (!nMember || !size)
    {
        return NULL;
    }

    if (nMember != buffSize / size)
    {
        return NULL;
    }

    ptr = malloc(buffSize);
    if (!ptr)
    {
        return NULL;
    }

    memset(ptr, 0, buffSize);

    return ptr;
}

                \end{align*}
            }
            \item
            {
                最小规格数： $V = 2^{1 - Bias}$ 。
                倒数： $V = 2^{Bias - 1}$ 。

                    M &= 0 \\
                    f &= 1 \\
                    V &= 0 1111...101 000000
                \end{align*}
            }
        \end{enumerate}
