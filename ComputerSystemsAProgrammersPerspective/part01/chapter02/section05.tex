void *calloc(size_t nMember, size_t size)
{
    size_t buffSize = nMember * size;
    void *ptr = NULL;

    if (!nMember || !size)
    {
        return NULL;
    }

    if (nMember != buffSize / size)
    {
        return NULL;
    }

    ptr = malloc(buffSize);
    if (!ptr)
    {
        return NULL;
    }

    memset(ptr, 0, buffSize);

    return ptr;
}

                    n &= 0.yyyyyyy\cdots \\
                    n << k &= y.yyyyyyy\cdots \\
                    n << k &= Y + n \\
                    n * 2^k - n &= Y \\
                    n &= Y / (2^k - 1)
                \end{align}
            }
            \item
            {
                \begin{enumerate}[(a)]

                    \item $n = 6 / (2^4 - 1) = 6 / 15 = 0.4$
                    \item $n = 19 / 31$
                \end{enumerate}
